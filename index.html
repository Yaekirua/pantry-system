Cengal 4 Pantry Management System

This is the official repository page for the Cengal 4 Pantry console application. The C++ source code below handles product registration, inventory tracking, payment processing with membership discounts, and daily sales reporting.

void registerProduct(int productid[], double productcost[], double productsellPrice[], int productquantity[],int totalquantity[], string productname[], char producttype[], int &productCount) 
{
        
    char reg = 'y';
    
    while (reg == 'Y' || reg == 'y') 
    {
        cout << "\t\t=========================================" << endl;
        cout << "\t\t\t      Register Product" << endl;
        cout << "\t\t=========================================\n" << endl;
    
        if (productCount >= MAX_PRODUCTS)                               //if product has reach the maximum
        {
            cout << "Maximum product capacity reached!" << endl;
            system("Pause");
            reg ='N';
        }
        else
        {
            cout << "\tEnter product ID: ";                         //enter product information
            cin >> productid[productCount];
            cin.ignore();
            cout << "\tEnter product name: ";
            getline(cin, productname[productCount]);
            cout << "\tEnter product type code (F for food or B for beverage): ";
            cin>>producttype[productCount];
            cout << "\tEnter product cost: ";
            cin >> productcost[productCount];
            cout << "\tEnter product sell price: ";
            cin >> productsellPrice[productCount];
            cout << "\tEnter product quantity: ";
            cin >> productquantity[productCount];
            totalquantity[productCount]=totalquantity[productCount]+productquantity[productCount];
            cout << "\n\t>>>>>> Product successfully registered!" << endl;
            productCount++;
    
            cout << "\n\tDo you want to register another product? (Y/N): ";         //to register more products
            cin >> reg;
            while (reg != 'Y' && reg != 'y' && reg != 'N' && reg != 'n') 
            {
                cout << "\n\tInvalid input. Please enter Y or N: ";
                cin >> reg;
            }
            system ("cls");
        }
    }
}

double payment(double productsellPrice[], string productname[], char producttype[], int productCount, double &totalToPay, int &totalProductsSold, int soldquantity[]) 
{
    double totalAmount=0.0, amountPaid, totalFoodPrice=0.0, totalBevPrice=0.0,discount=0.0;
    totalToPay=0.0;
    char toBuy, more,member;
    int foodChoice, bevChoice, quantity;
    
    
    do
    {
    cout << "\t\t=========================================" << endl;      //function to buy and make payment
    cout << "\t\t\t      Buy & Payment " << endl;
    cout << "\t\t=========================================\n" << endl;
    
    cout<<"\t\tFood or Beverages? (F / B):"<<endl;
    cout<<"\t\t";
    cin>>toBuy;
    
    
    if (toBuy == 'f'|| toBuy =='F')
    {
    
        cout<<"\n\t\t-----------------------------------------------"<<endl;     //display list of foods
        cout<<"\t\t\tFood\t\t\tPrice"<<endl;
        cout<<"\t\t-----------------------------------------------"<<endl;
        
        for (int i = 0; i < productCount; i++) 
        {
            if (producttype[i]== 'f' || producttype[i]== 'F')
                cout << "\t\t" <<(i+1)<<". "<< productname[i] <<"\t\t\t RM" << productsellPrice[i] << endl;
        }
        cout << "\t\t----------------------------------------------" << endl;
    
    
    cout<<"\t\tEnter food choice:";
    cin>>foodChoice;
    cout<<"\t\tEnter quantity:";
    cin>>quantity;
    
    soldquantity[foodChoice-1] += quantity;                                 //formula for important calculations
    totalProductsSold +=quantity;
    totalFoodPrice = productsellPrice[foodChoice - 1] * quantity;
    cout<<"\t\t>>>> Total price: "<<totalFoodPrice<<endl;
    
    totalToPay = totalToPay + totalFoodPrice;
    }
    else if (toBuy == 'b'|| toBuy =='B')
    {
    
        cout<<"\t\t-----------------------------------------------"<<endl;   //display list of beverages
        cout<<"\t\t\tBeverage\t\t\tPrice"<<endl;
        cout<<"\t\t-----------------------------------------------"<<endl;
        
        for (int i = 0; i < productCount; i++) 
        {
            if (producttype[i]== 'b' || producttype[i]== 'B')
                cout << "\t\t" <<(i+1)<<". "<< productname[i] <<"\t\t\t RM" << productsellPrice[i] << endl;
        }
        cout << "\t\t---------------------------------------------" << endl;
    
    
    cout<<"\t\tEnter beverage choice:";
    cin>>bevChoice;
    cout<<"\t\tEnter quantity:";
    cin>>quantity;
    
    soldquantity[bevChoice-1] += quantity;
    totalBevPrice = productsellPrice[bevChoice - 1] * quantity;
    totalProductsSold +=quantity;
    cout<<"\t\tTotal Price: RM: "<<totalBevPrice<<endl;
    totalToPay = totalToPay + totalBevPrice;
    }
    
    
    cout<<"\n\t\tGrand Total: RM"<<totalToPay<<endl;
    
    cout<<"\n\t\tDo you want to buy another food or beverages? (Y / N): ";  //to buy more food/beverages
    cin>>more;
    
    system ("cls");
    }while (more == 'y' || more == 'Y');
    
    cout<<"\t\t---------------------------------------------" << endl;        
    cout<<"\t\t\tPAYMENT"<<endl;
    cout<<"\t\t---------------------------------------------" << endl;
    
    cout<<"\n\n\t\tMember or non member[M/N]?:"<<endl;                          //checking for membership status
    cin>>member;
        
    cout<<"\n\n\t\t---------------------------------------------" << endl;        //display receipt
    cout<<"\t\t\tRECEIPT"<<endl;
    cout<<"\t\t---------------------------------------------" << endl;
    cout<<"\n\t\tTotal Price\t\t: RM"<<totalToPay<<endl;
    if (member=='m'||member=='M')
        {
            discount=totalToPay * 0.10;                                     //member will get 10% discounts
            totalToPay = totalToPay * 0.90;
            cout<<"\t\tDiscount 10%\t\t: RM"<<discount<<endl;
        }
    cout<<"\n\t\tPrice After Discount\t: RM"<<totalToPay<<endl;
    cout<< "\t\tEnter amount paid\t: RM";
    cin>> amountPaid;
    
    if (amountPaid > totalToPay) 
    {
        cout<<"\n\t\tBalance\t\t\t: RM"<<amountPaid - totalToPay<<endl;
        cout<<"\n\t\t";
        system("Pause"); 
        return amountPaid - totalAmount;   
    } 
    else if (amountPaid == totalToPay) 
    {
        cout<<"\n\t\tNo Balance."<<endl;
        cout<<"\n\t\t";
        system("Pause"); 
        return amountPaid - totalToPay;    
    }
    else 
    {
        cout << "\n\t\tInsufficient payment. Please give sufficient value." << endl;
        cout<<"\n\t\t";
        system("Pause"); 

        return 0.0;
    }
    
}

void dailyReport(int totalProductsSold, double totalToPay,int totalquantity[],int soldquantity[],int productid[], double productcost[], double productsellPrice[], int productquantity[], string productname[],int productCount,char producttype[]) 
{
    
    string preferFood, preferBev, dislikeFood, dislikeBev;
    int maxFood = -99, maxBev = -99, minFood= 100, minBev=100;
    
    cout << "\t\t=========================================" << endl;  //funstion to display  daily report
    cout << "\t\t\t      Daily Report " << endl;
    cout << "\t\t=========================================\n" << endl;
    
   time_t now = time(0);                                                //to get and display curret date 
   tm *ltm = localtime(&now); 
   cout << "\t\tDate : "<< ltm->tm_mday<<" / "<< 1 + ltm->tm_mon<<" / "<< 1900 + ltm->tm_year<<endl;   
        
    cout << "\nID\tFood\t\t   Cost(RM)\tSell Price(RM)\t Quantity In\t Quantity Sold\tBalance in stock"<<endl;
    cout<<"--------------------------------------------------------------------------------------------------------------" << endl;
    
    for (int i = 0; i < productCount; i++) 
    {
        
        if (producttype[i]=='f' || producttype[i]== 'F')
        {
            cout << productid[i]
        << "\t" << productname[i]
        << "\t\t" << productcost[i]
        << "\t\t" << productsellPrice[i]
        << "\t\t" << totalquantity[i]
        << "\t\t"<<soldquantity[i]
        << "\t\t"<<(totalquantity[i]-soldquantity[i])<<endl;
        }
    }
    cout << "---------------------------------------------------------------------------------------------------------------" << endl;
    cout << "\nID\tBeverage\t  Cost(RM)\tSell Price(RM)\t Quantity In\t Quantity Sold\tBalance in stock"<<endl;
    cout<<"--------------------------------------------------------------------------------------------------------------" << endl;
    for (int i = 0; i < productCount; i++) 
    {
        
        if ((producttype[i]=='b' || producttype[i]== 'B'))
        {
            cout << productid[i]
        << "\t" << productname[i]
        << "\t\t" << productcost[i]
        << "\t\t" << productsellPrice[i]
        << "\t\t" << totalquantity[i]
        << "\t\t"<<soldquantity[i]
        << "\t\t"<<(totalquantity[i]-soldquantity[i])<<endl;
            }
}
    for (int i = 0; i < productCount; i++)                                  //to find the highest sales of food
    {
        
        if ((producttype[i]=='f' || producttype[i]== 'F')&& soldquantity[i]> maxFood)
        {
            maxFood = soldquantity[i];
            preferFood = productname[i];
        }
    }
    for (int i = 0; i < productCount; i++)                                  //to find the highest sales of beverages
    {
        
        if ((producttype[i]=='b' || producttype[i]== 'B') && soldquantity[i]> maxBev)
        {
            maxBev = soldquantity[i];
            preferBev = productname[i];
        }
    }
        
    for (int i = 0; i < productCount; i++)                                  //to find the lowest sales of food
    {
        if ((producttype[i]=='f' || producttype[i]== 'F') && soldquantity[i]< minFood)
        {
            minFood = soldquantity[i];
            dislikeFood = productname[i];
        }
    }
    
    for (int i = 0; i < productCount; i++)                                  //to find the lowest sales of beverages
    {
        if ((producttype[i]=='b' || producttype[i]== 'B') && soldquantity[i]< minBev)
        {
            minBev = soldquantity[i];
            dislikeBev = productname[i];
        }
    }
        
    cout << "---------------------------------------------------------------------------------------------------------------" << endl;
    cout << "\n\tTotal products sold: " << totalProductsSold << endl;
    cout << "\tTotal sale: RM" << totalToPay << endl;
    cout << "\n\n\t------- Analysis---------"<<endl;
    cout << "\tFood with highest sales: "<<preferFood<<endl;
    cout << "\tBeverages with highest sales: "<<preferBev<<endl;
    cout << "\n\tFood with lowest sales: "<<dislikeFood<<endl;
    cout << "\tBeverages with lowest sales: "<<dislikeBev<<endl;
    cout << "\n\t";
    system("Pause");
}
